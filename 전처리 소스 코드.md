import pandas as pd
import re
import os

def clean_email_text(text):
"""
[기획서 8.3 준수] 이메일 본문(Raw Text) 정제 및 일반화 모듈
- 목적: NLP 모델의 노이즈 처리 부담을 줄이고 문맥 파악 효율을 극대화함
- 처리 로직:
1. HTML 태그/스크립트 제거 (XSS 공격 코드 포함)
2. 하이퍼링크 마스킹 ([URL] 토큰 치환) - 개별 URL에 대한 과적합(Overfitting) 방지
3. 비표준 특수문자 제거 및 다중 공백 단일화
"""
if pd.isna(text):
return ""

```
text = str(text)
# 1. HTML 태그 제거
text = re.sub(r'<[^>]+>', ' ', text)
# 2. URL 마스킹 처리
text = re.sub(r'http[s]?://\\S+', '[URL]', text)
# 3. 화이트리스트 기반 특수문자 필터링 (한/영/숫자 및 기본 문장부호 유지)
text = re.sub(r'[^a-zA-Z0-9가-힣.,!?\\s]', ' ', text)

# 4. 규격화된 공백 처리
return re.sub(r'\\s+', ' ', text).strip()
```

print("[System] 기획서 8.1 기반 이메일 마스터 데이터 병합 프로세스 가동...")
email_master_list = []

# [스키마 정의] 파일별 본문 Feature명 매핑 (라벨은 전역적으로 'label' 고정 확인됨)

email_source_map = {
'Enron.csv': 'body',
'SpamAssassin.csv': 'body',
'phishing_email.csv': 'text_combined',
'Nazario.csv': 'body',
'Nazario_5.csv': 'body',
'CEAS_08.csv': 'body',
'Ling.csv': 'body',
'Nigerian_Fraud.csv': 'body'
}

for file, content_col in email_source_map.items():
if os.path.exists(file):
try:
# 메모리 최적화를 위해 필요한 컬럼만 로드 및 결측치 즉시 제거
df = pd.read_csv(file, usecols=[content_col, 'label']).dropna()

```
        # 피처 스키마 통합 (Input: text, Target: label)
        df.rename(columns={content_col: 'text'}, inplace=True)

        # 라벨 데이터 정규화 (전처리 표준: 0=정상, 1=피싱)
        df['label'] = pd.to_numeric(df['label'], errors='coerce').fillna(0).astype(int)
        email_master_list.append(df)
        print(f"  - [Success] {file} 로드 완료")
    except Exception as e:
        print(f"  - [Error] {file} 처리 중 예외 발생: {e}")
```

if email_master_list:
# 수집된 모든 코퍼스 수직 병합
email_master = pd.concat(email_master_list, ignore_index=True)

```
# 기획서 8.3 공정 이행: 벡터화 연산을 통한 텍스트 세척 수행
print("  - 텍스트 세척 및 토큰 표준화 진행 중...")
email_master['cleaned_text'] = email_master['text'].apply(clean_email_text)

# 최종 NLP 학습용 데이터셋 익스포트
email_master[['cleaned_text', 'label']].to_csv('NLP_MASTER_DATA.csv', index=False)
print(f"✅ NLP_MASTER_DATA.csv 발행 완료 (총 {len(email_master)}행)")
```

# Phase 2: URL 탐지(ML) 데이터셋 통합 파이프라인

print("\n[System] 기획서 8.2 기반 URL 마스터 데이터 병합 프로세스 가동...")
url_master_list = []

# 1. malicious_phish.csv (PhishTank/Kaggle 기반 대용량 세트)

if os.path.exists('malicious_phish.csv'):
df_u1 = pd.read_csv('malicious_phish.csv', usecols=['url', 'type']).dropna()
# 이진 분류를 위한 라벨 인코딩 (benign=0, others=1)
df_u1['label'] = df_u1['type'].apply(lambda x: 0 if str(x).strip().lower() == 'benign' else 1)
url_master_list.append(df_u1[['url', 'label']])
print("  - malicious_phish.csv 로드 및 라벨링 완료")

# 2. PhiUSIIL_Phishing_URL_Dataset.csv (UCI/연구용 최신 세트)

if os.path.exists('PhiUSIIL_Phishing_URL_Dataset.csv'):
df_u2 = pd.read_csv('PhiUSIIL_Phishing_URL_Dataset.csv', usecols=['URL', 'label']).dropna()
df_u2.rename(columns={'URL': 'url'}, inplace=True)

```
# [주의] PhiUSIIL 데이터셋의 특이 라벨링 교정 (원본 1:정상 -> 0, 0:피싱 -> 1)
df_u2['label'] = df_u2['label'].apply(lambda x: 1 if x == 0 else 0)
url_master_list.append(df_u2)
print("  - PhiUSIIL 데이터셋 로드 및 라벨 반전 교정 완료")
```

if url_master_list:
# URL 마스터 셋 최종 병합
url_master = pd.concat(url_master_list, ignore_index=True)
url_master.to_csv('URL_MASTER_DATA.csv', index=False)
print(f"✅ URL_MASTER_DATA.csv 발행 완료 (총 {len(url_master)}행)")
